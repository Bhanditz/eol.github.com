173:   def page
174:     # get the id parameter, which can be either a page ID # or a page name
175:     @page_id=params[:id]
176: 
177:     raise "static page without id" if @page_id.nil? || @page_id==''
178:     
179:     unless read_fragment(:controller=>'content',:part=>@page_id + "_" + current_user.language_abbr)
180:         # if the id is not numeric, assume it's a page name
181:         if @page_id.to_i == 0 
182:           page_name=@page_id.gsub(' ','_').gsub('_',' ')
183:           @content=ContentPage.get_by_page_name_and_language_abbr(page_name,current_user.language_abbr)
184:         else # assume the id passed is numeric and find it by ID
185:           @content=ContentPage.get_by_id_and_language_abbr(@page_id,current_user.language_abbr)
186:         end
187:         
188:         raise "static page content not found" if @content.nil?
189:         
190:         # if this static page is simply a redirect, then go there
191:         if !@content.url.blank?
192:           headers["Status"] = "301 Moved Permanently"
193:           redirect_to(@content.url)
194:         end
195:         
196:     end
