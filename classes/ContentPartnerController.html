171:   def agreement
172:     
173:     # TO UPDATE THE CONTENT PARTNER AGREEMENT TEMPLATE AND ENSURE THAT EACH PREVIOUS CONTENT PARTNER GETS A NEW ONE WITH THE NEWLY UPDATED TEMPLATE,
174:     # SET THEIR CURRENT CONTENT PARTNER AGREEMENT TO "IS_CURRENT=FALSE" --- This is done automatically when using the admin interface to edit
175:     agreement_id=params[:agreement_id] || ""
176:     
177:     # if we are calling this method from the content partner registry, the agent is currently logged in so use that one
178:     if agent_logged_in? && params[:id].nil?
179:       @agent=current_agent
180:     else #otherwise, show the agreement from the agent ID passed into the querystring (and the specific agreement if passed in)
181:       @agent=Agent.find(params[:id])
182:     end
183:     
184:     unless @agent.ready_for_agreement?
185:       flash[:warning] = "The agreement for this partner is not available."
186:       redirect_to(:action => 'index')
187:       return
188:     end
189:     
190:     # find their agreement
191:     if agreement_id.empty? 
192:       @agreement=@agent.agreement      
193:       # if this is the first time they are viewing the agreement, create it from the default template
194:       @agreement=ContentPartnerAgreement.create_new(:agent_id=>@agent.id) if @agreement.nil?
195:       # update the number of views if the content partner is viewing it
196:       @agreement.update_attributes(:number_of_views=>@agreement.number_of_views+=1,:last_viewed=>Time.now) if !current_agent.nil?
197:     elsif current_user.is_admin?
198:       @agreement=ContentPartnerAgreement.find_by_id_and_agent_id(params[:agreement_id],@agent.id,:order=>'created_at DESC')
199:       @agreement=ContentPartnerAgreement.create_new(:agent_id=>@agent.id) if @agreement.nil?
200:     end
201:     
202:     @primary_contact=@agent.primary_contact  
203:     
204:     if !@agreement.mou_url.blank?  # if there is a URL, render the url 
205:       redirect_to @agreement.mou_url
206:     else #otherwise render the template
207:       render :layout => false, :inline=>@agreement.template
208:     end
209:     
210:   end
